pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Deymonryk/farm_breakdown.git'
            }
        }
        
	stage('Build') {
            steps {
		bat "\"${tool 'MSBuild'}\" /p:Configuration=Debug /t:Build Farm_breakdown.sln"
                bat 'echo "Building"'
            }
        }

        stage('Test') {
            steps {
                 bat 'echo "Run"'
            }
        }
        
    }
}
