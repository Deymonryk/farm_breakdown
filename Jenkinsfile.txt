    agent any
    
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Deymonryk/farm_breakdown.git'
            }
        }
        
	stage('Build') {
            steps {
                sh 'mkdir build && cd build && cmake .. && make'
            }
        }

        stage('Test') {
            steps {
                sh 'cd build && ctest'
            }
        }
        
    }
}
